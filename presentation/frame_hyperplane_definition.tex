\begin{frame}
	\frametitle{Definition}
	\begin{block}{Geometric hyperplane}
		An hyperplane $H$ is a proper subspace of $S_k(3)$ such that for each line of $S_k(3)$, $H$ either contain the full line or only one point of the line.
	\end{block}
